━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. ОТКРОЙТЕ файл .env
2. НАЙДИТЕ строку: ADMIN_IDS=ВАШ_ID_СЮДА
3. ЗАМЕНИТЕ "ВАШ_ID_СЮДА" на ваш Telegram ID
4. КАК УЗНАТЬ ID:
   - Telegram → @userinfobot
   - Нажмите Start
   - Скопируйте число из "Id: XXXXXXX"
5. СОХРАНИТЕ файл .env

Пример правильного .env:
ADMIN_IDS=7889072630

2. File Manager → Upload Files
3. Выберите 4 файла:
   - main.py
   - dota2_real_lobby_bot.py
   - requirements.txt
   - .env (с ВАШИМ ID!)
4. START сервера
5. Проверьте логи
6. Telegram → /start
